DROP DATABASE IF EXISTS STUDENTS;
CREATE DATABASE STUDENTS;
USE STUDENTS;

DROP TABLE IF EXISTS Student ; 
CREATE TABLE Student (id_student INT AUTO_INCREMENT NOT NULL, stu_name VARCHAR(50), stu_first_name VARCHAR(50), stu_date_birth DATE, stu_adress_resid VARCHAR(50), stu_univ_mail VARCHAR(50), stu_phone_num VARCHAR(50), stu_gender VARCHAR(10), stu_nationality VARCHAR(50), stu_current_level VARCHAR(25), id_department INT,
 id_accomodation INT,
  id_scholarship INT,
  id_internship INT,
  id_job INT,
  id_relation_inter INT,
  id_insertion INT,
  PRIMARY KEY (id_student)); 

DROP TABLE IF EXISTS Course ; 
CREATE TABLE Course (id_course INT AUTO_INCREMENT NOT NULL, course_title VARCHAR(50), disciplinary_area VARCHAR(50), credits_ECTS INT, start_time_course TIME, end_time_course TIME, number_hours float (4,2), day_of_course VARCHAR(50), method_delivery_course VARCHAR(50), type_course VARCHAR(50), PRIMARY KEY (id_course)); 

DROP TABLE IF EXISTS Professor ; 
CREATE TABLE Professor (id_prof INT AUTO_INCREMENT NOT NULL, prof_name VARCHAR(50), first_name_prof VARCHAR(50), e_mail_prof VARCHAR(50), phone_num_prof VARCHAR(50), expertise_field VARCHAR(50), availability_prof VARCHAR(50), office_adress VARCHAR(50), department_prof VARCHAR(50), status_prof VARCHAR(25), PRIMARY KEY (id_prof));  


DROP TABLE IF EXISTS Registrations ; 
CREATE TABLE Registrations (id_registrations INT AUTO_INCREMENT NOT NULL, id_student INT NOT NULL, id_course INT NOT NULL, semester_regist INT, year_integration YEAR, status_registration VARCHAR(50), date_regist DATE, univ_component VARCHAR(50), sector_regist VARCHAR(50), id_department INT, PRIMARY KEY (id_registrations), 
FOREIGN KEY (id_student) REFERENCES Student (id_student) ON DELETE CASCADE, FOREIGN KEY (id_course) REFERENCES course (id_course) ON DELETE CASCADE) ;  


DROP TABLE IF EXISTS Notes ; 
CREATE TABLE Notes (id_student INT NOT NULL, id_course INT NOT NULL, mean_semester float(4,2), year_graduation YEAR, type_evaluation VARCHAR(50), mention VARCHAR(50), validity_year VARCHAR(50), rank_student VARCHAR(50), PRIMARY KEY (id_student, id_course), FOREIGN KEY (id_student) REFERENCES student (id_student) ON DELETE CASCADE , FOREIGN KEY (id_course) REFERENCES course (id_course) ON DELETE CASCADE);  

DROP TABLE IF EXISTS Departments ; 
CREATE TABLE Departments (id_department INT AUTO_INCREMENT NOT NULL, dep_name VARCHAR(50), dep_manager VARCHAR(50), localisation_dep VARCHAR(50), phone_num_dep VARCHAR(50), email_dep VARCHAR(50), budget_dep VARCHAR(50), description_dep VARCHAR(50), number_employees INT, number_student_per_dep INT, id_faculty INT, PRIMARY KEY (id_department)) ;  

DROP TABLE IF EXISTS Student_services ; 
CREATE TABLE Student_services (id_service INT AUTO_INCREMENT NOT NULL, name_service VARCHAR(50), description_service VARCHAR(50), contact_detail VARCHAR(50), Business_hours VARCHAR(50), Responsible VARCHAR(50), Type_service VARCHAR(50), budget VARCHAR(50), PRIMARY KEY (id_service)) ;  


DROP TABLE IF EXISTS Student_housing ; 
CREATE TABLE Student_housing (id_accomodation INT AUTO_INCREMENT NOT NULL, type_accomodation VARCHAR(50), area VARCHAR(50), availability VARCHAR(50), location VARCHAR(50), price VARCHAR(50), service_included VARCHAR(50), time_accomodation VARCHAR(50), rental_start DATE, rental_end_date DATE, id_student INT, PRIMARY KEY (id_accomodation));  


DROP TABLE IF EXISTS campus_jobs ;
CREATE TABLE campus_jobs (id_job INT AUTO_INCREMENT NOT NULL, position_title VARCHAR(50), description VARCHAR(50), workplace VARCHAR(50), working_hours VARCHAR(50), salary VARCHAR(50), supervisor VARCHAR(50), start_date DATE, end_date DATE, employment_status VARCHAR(50), id_student INT, PRIMARY KEY (id_job)) ;  


DROP TABLE IF EXISTS Scholarships ; 
CREATE TABLE Scholarships (id_scholarship INT AUTO_INCREMENT NOT NULL, scholarship_name VARCHAR(50), description VARCHAR(50), eligibility_criteria VARCHAR(50), amount VARCHAR(50), duration VARCHAR(50), application_deadline VARCHAR(50), selection_process VARCHAR(50), previous_beneficiaries VARCHAR(50), status VARCHAR(50), id_student INT, PRIMARY KEY (id_scholarship)) ;  


DROP TABLE IF EXISTS Sports_disciplines ; 
CREATE TABLE Sports_disciplines (id_sport_disc INT AUTO_INCREMENT NOT NULL, name_discipline VARCHAR(50), description VARCHAR(50), equipement_available VARCHAR(50), hours_operation VARCHAR(50), registration_date DATE, additional_service VARCHAR(50), membership_rates VARCHAR(50), special_events VARCHAR(50), discipline_teacher VARCHAR(50), PRIMARY KEY (id_sport_disc)) ;  


DROP TABLE IF EXISTS Calendar ;
CREATE TABLE Calendar (id_calendar INT AUTO_INCREMENT NOT NULL, id_course INT NOT NULL, class_period VARCHAR(50), holiday_period VARCHAR(50), review_period VARCHAR(50), courses_times VARCHAR(50),  public_holidays VARCHAR(50), PRIMARY KEY (id_calendar), FOREIGN KEY (id_course) REFERENCES course (id_course) ON DELETE CASCADE) ; 

DROP TABLE IF EXISTS Exam_assessment ;
CREATE TABLE Exam_assessment (id_exam INT AUTO_INCREMENT NOT NULL, id_course INT NOT NULL, matter VARCHAR(50), type_exam VARCHAR(50), duration_assessment VARCHAR(50), examination_room VARCHAR(50), number_student INT, results VARCHAR(50), PRIMARY KEY (id_exam),FOREIGN KEY (id_course) REFERENCES Course (id_course) ON DELETE CASCADE) ;  

DROP TABLE IF EXISTS Tutoring_mentoring ; 
CREATE TABLE Tutoring_mentoring (id_mentoring INT AUTO_INCREMENT NOT NULL, type_program VARCHAR(50), mentor_field_study VARCHAR(50), availability_tutors VARCHAR(50), number_stud_tutors VARCHAR(50), id_student INT, tutoring_session VARCHAR(50), program_objective VARCHAR(50), contact_mentors VARCHAR(50), PRIMARY KEY (id_mentoring), FOREIGN KEY (id_student) REFERENCES student (id_student) ON DELETE CASCADE) ; 


DROP TABLE IF EXISTS International_relationship ; 
CREATE TABLE International_relationship (id_relation_inter INT AUTO_INCREMENT NOT NULL, exchange_program VARCHAR(50), international_partnership VARCHAR(50), id_student INT, partner_university VARCHAR(50), country VARCHAR(50), program_duration VARCHAR(50), main_contact VARCHAR(50), PRIMARY KEY (id_relation_inter),FOREIGN KEY (id_student) REFERENCES student (id_student) ON DELETE CASCADE) ;  


DROP TABLE IF EXISTS Alumni_network ; 
CREATE TABLE Alumni_network (id_alumni INT AUTO_INCREMENT NOT NULL, name_association VARCHAR(50), objective_association VARCHAR(50), proposed_service VARCHAR(50), alumni_stud VARCHAR(50), membership_codition VARCHAR(50), member_benefit VARCHAR(50), contacts VARCHAR(50), PRIMARY KEY (id_alumni)) ;  


DROP TABLE IF EXISTS Professional_integration ;
CREATE TABLE Professional_integration (id_insertion INT AUTO_INCREMENT NOT NULL, graduation_year VARCHAR(50), degree_obtained VARCHAR(50), latency_time_before_insertion VARCHAR(50), id_student INT, salary_obtaines VARCHAR(50), type_contract_obtaines VARCHAR(50), PRIMARY KEY (id_insertion),FOREIGN KEY (id_student) REFERENCES student (id_student) ON DELETE CASCADE) ;  


DROP TABLE IF EXISTS Internship ; 
CREATE TABLE Internship (id_internship INT AUTO_INCREMENT NOT NULL, id_student INT, duration_internship VARCHAR(25), enterprise_name VARCHAR(50), enterprise_location VARCHAR(50), PRIMARY KEY (id_internship),FOREIGN KEY (id_student) REFERENCES student (id_student) ON DELETE CASCADE) ;  


DROP TABLE IF EXISTS Campus ; 
CREATE TABLE Campus (id_campus INT AUTO_INCREMENT NOT NULL, id_department INT, camp_location VARCHAR(50), camp_components VARCHAR(50), PRIMARY KEY (id_campus),FOREIGN KEY (id_department) REFERENCES departments (id_department) ON DELETE CASCADE);  


DROP TABLE IF EXISTS Borrowed_books_lib ;

DROP TABLE IF EXISTS Student_transport ;
